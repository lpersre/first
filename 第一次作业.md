```C++
#include<iostream>  
using namespace std;  
  
int main()  
{  
    int a[9];  
    cout << "随机输入9个数" << endl;  
    for(int i = 0; i < 9; i++)   
    {  
        cin >> a[i];  
    }
//冒泡排序
    for (int i = 0; i < 8; i++) //每次外循环将本次循环最大数排至末尾
    {  
        for (int j = 0; j < 8 - i; j++)  
        {  
            if(a[j] > a[j + 1])   
            {  
                int t = a[j];  
                a[j] = a[j + 1];  //冒泡排序，大数置后
                a[j + 1] = t;  
            }  
        }  
    }  
    for(int i = 0; i < 9; i++)  
    {  
        cout << a[i] << "\n";   
    }  
    return 0;  
}
```

```C++
#include<iostream>
```

```C++
int main()
{
    int m,n,g,l;
    int m2,n2;
    cin>>m>>n;//输入原始数据
    int t,t2;
    m2=m,n2=n;//复制一份原始数据用于辗转相除
    if(m2<n2)//确保大数为被除数
    {
        t=m2;
        m2=n2;
        n2=t;
    }
    while(m2%n2!=0)//当余数不为0时辗转相除
    {
        t2=m2%n2;
        m2=n2;
        n2=t2;
    }
    g=n2;//最大公约数
    l=(m*n)/g;//最小公倍数
    cout<<g<<"\n";
    cout<<l<<"\n";
    return 0;
}
```

第八题
```C++
#include <iostream>
using namespace std;
int main() {
    int arr[10] = {0}; // 预留空间给新元素
    int newElement; // 存储需要插入的新元素
    int size = 9; // 原始数组的大小
    // 读取原始数组
    for (int i = 0; i < size; ++i) {
        cin >> arr[i];
    }
    // 读取需要插入的新元素
    cin >> newElement;
    // 找到新元素应该插入的位置
    int i;
    for (i = size; (i > 0) && (arr[i - 1] > newElement); --i) {
        arr[i] = arr[i - 1]; // 向后移动元素
    }
    // 插入新元素
    arr[i] = newElement;
    size++; // 更新数组大小
    for (i= 0; i < size; ++i) {
        cout << arr[i] << endl;
    }
    return 0;
}
```

#include <iostream>
using namespace std;
void reverse(int* a1, int array_length) {
    int* start = a1;          // 指向数组的开始
    int* end = a1 + array_length - 1; // 指向数组的结束
    while (start < end) //冒泡排序
    {
        int temp = *start;
        *start = *end;
        *end = temp;
        // 移动指针
        start++;
        end--;
    }
}
int main() {
    int n = 10;
    int a[n];
     // 数组元素个数为10
    // 读取10个整数到数组a中
    cout << "Enter 10 个int数:" << endl;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
//引用函数
    reverse(a, n);
    cout << "The array has been reverted:" << endl;
    for (int i = 0; i < n; i++) {
        cout << a[i] << "\n";
    }
    return 0;
}
